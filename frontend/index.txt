<script>
async function ask() {
  const q = document.getElementById("q").value;
  const out = document.getElementById("out");
  out.innerHTML = "Thinking...";

  const res = await fetch(
    `http://127.0.0.1:8000/ask?question=${encodeURIComponent(q)}`
  );
  const data = await res.json();

  let html = `<b>Answer:</b><br>${data.answer}`;

  if (data.sources) {
    html += "<hr><b>Sources:</b><ul>";
    data.sources.forEach(s => {
      html += `<li><a href="${s.link}" target="_blank">${s.title}</a></li>`;
    });
    html += "</ul>";
  }
  out.innerHTML = html;
}
</script>
